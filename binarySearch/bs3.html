                f,fi:array[0..1000000] of longint;
        Function Tim (l,r,i:int64):int64;
        var	mid:longint;
        begin
          while (l<=r) do begin
            mid:=(l+r) div 2;
            if (f[mid-1]<i) and (f[mid]>=i) then exit(mid) else
            if (f[mid]>i) then r:=mid-1 else l:=mid+1;
          end;
        end;

          readln(n);
          f[0]:=0;
          for i:=1 to n do begin

          for i:=1 to m do begin
            read(b[i]);
