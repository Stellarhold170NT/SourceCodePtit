              USES CRT;
        var     n,i:int32;
                res:int64;
                a,dem:array[1..trunc(1e5)] of longint;
                begin
                    readln(n);
                    for i:=1 to n do begin
                        read(a[i]);
                        inc(dem[a[i]]);
                        if (dem[a[i]]=2) then res:=res+1 else
                        if (dem[a[i]]>2) then res:=res+(dem[a[i]]-1);
                    end;
                    writeln(res);
                end.
