begin
    i:=l;
    j:=r;
    x:=a[(l+r) div 2];
    repeat
        while a[i]>x do inc(i);
        while a[j]<x do dec(j);
        if not (i>j) then begin
            tg:=a[i];
            a[i]:=a[j];
            a[j]:=tg;
            inc(i);
            dec(j);
        end;
    until i>j;
    if j>l then SORT(a,l,j);
    if i<r then SORT(a,i,r);
end;
