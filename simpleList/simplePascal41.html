              Uses Crt;
        Var     c,l,a:array[0..1000000] of int64;
                n,i:longint;
                kq:int64;
        Begin
            readln(n);
            c[0]:=0; l[0]:=0; c[1]:=0; l[1]:=0;
            for i:=1 to n do
              begin
                read(a[i]);
                if a[i] mod 2=0 then c[i]:=c[i-1]+1 else c[i]:=c[i-1];
                if a[i] mod 2<>0 then l[i]:=l[i-1]+1 else l[i]:=l[i-1];
              end;
            kq:=0;
            //writeln(l[n]);
            if l[n] mod 2<>0 then
            kq:=l[n]*c[n];
        
            if l[n] mod 2=0 then
            kq:=kq+(l[n]*(l[n]-1)) div 2+(c[n]*(c[n]-1)) div 2;
        
            writeln(kq);
            readln;
        
        End.
