              Uses Crt;
        Var n,v,i,j,k,vt1,vt,kq,kq1:longint; str,r:string;
        Begin
            readln(n);
            readln(str);
            readln(v,r);
            delete(r,1,1);
            insert(r,str,v);
            j:=v;
            kq:=0;
            repeat
                kq1:=0;
                if (str[j]=str[j-1]) or (kq=0) then
                begin
                    kq1:=kq1+1;
                    for k:=j downto 1 do
                     if str[k]=str[k-1] then inc(kq1)
                     else begin vt:=k; break; end;
                     
                    for k:=j to length(str) do
                     if str[k]=str[k+1] then inc(kq1)
                     else begin vt1:=k; break; end;
                     
                    if kq1>=3 then kq:=kq+kq1 else break;
                    delete(str,vt,vt1-vt+1);
                    j:=vt;
                end
                else break;
            until false;
            writeln(kq);
        End.
