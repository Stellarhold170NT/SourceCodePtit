          int t;

          for (int i = 1; i <= t; i++) {
            int x1, y1, x2, y2;
            scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
        
            if (abs(x2 - x1) == abs(y2 - y1)) {

            }
          }
