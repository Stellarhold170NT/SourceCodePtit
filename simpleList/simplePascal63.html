        Procedure Sort(l,r:longint);
        var i,j,x:longint;
        begin
            i:=l; j:=r; x:=a[(l+r) div 2];
            repeat
                while a[i]<x do inc(i); while (a[j]>x) do dec(j);
                if i<=j then begin Hoanvi(a[i],a[j]); hoanvi(b[i],b[j]); inc(i); dec(j) end;
            until i>j;
            if j>l then sort(l,j); if i<r then sort(i,r);
        end;
