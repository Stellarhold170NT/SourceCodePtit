              #include <iostream>
        using namespace std;
         
        int main() {
          int n, h;
          cin >> n >> h;
         
          int a[100001];
         
          int i = 1;
          while (n > 0) {
            a[i] = n % 2;
            n /= 2;
            i++;
          }
         
          for (int j = i - 1; j >= 1; j--) {
            cout << a[j];
          }
          cout << endl;
         
          return 0;
        }
