              Uses Crt;
        Var  n,i,j,k,vitri,res:longint;
             p,s:array[0..10000] of ansistring;
        Begin
            p[0]:='_0'; p[1]:='.,?!1'; p[2]:='abc2'; p[3]:='def3';
            p[4]:='ghi4'; p[5]:='jkl5'; p[6]:='mno6'; p[7]:='pqrs7';
            p[8]:='tuv8'; p[9]:='wxyz9';
            readln(n);
            for i:=1 to n do
            begin
                readln(s[i]);
                vitri:=1; res:=0;
                for j:=1 to length(s[i]) do
                begin
                    for k:=0 to 9 do
                    if pos(s[i][j],p[k])<>0 then
                    begin
                        inc(res,pos(s[i][j],p[k]));
                        break;
                    end;
                    if vitri<>k then
                    begin
                        inc(res,1);
                        vitri:=k;
                    end;
                end;
                writeln(res);
            end;
            readln;
        End.
