              Uses Crt;
        Var a:array[0..500001] of longint;
                n,k,i,count:longint;
        Begin
            readln(n);
            count:=0;
            k:=0;
            repeat
                inc(k);
                for i:=1 to k do
                begin
                        inc(count);
                        a[count]:=(a[count-1] div k+1)*k;
                        if count=n then break;
                end;
            until count>=n;
            writeln(a[n]);
            readln;
        End.
