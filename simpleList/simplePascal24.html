              Uses Crt;
        Var n,i,d:longint;
        Begin
            readln(n);
            for i:=2 to trunc(sqrt(n)) do begin
                d:=0;
                if n mod i=0 then begin
                    while n mod i=0 do begin
                        n:=n div i;
                        inc(d);
                        if n>1 then write(i,'*') else write(i);
                    end;
                    //if n>1 then write(i,'^',d,'*') else write(i,'^',d);
                end;
            end;
            if n>1 then write(n);
            readln;
        End.
