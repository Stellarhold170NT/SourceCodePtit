              Uses Crt,math;
        Var n,i,j:longint;
            kq:int64;
            a,b,d:array[-40..10000] of int64;
        Begin
            readln(n);
            //f[0]:=0;
            kq:=-maxlongint;
            for i:=1 to n do
            begin
                 read(a[i]);
                 b[i]:=b[i-1]+a[i];
            end;
            //for i:=1 to n do write(b[i],' ');      writeln;
            for i:=3 to n do
            begin
                d[i]:=max(b[i]-b[i-3],d[i-3]+b[i]-b[i-3]);
                if d[i]>kq then kq:=d[i];
                if d[i]<0 then d[i]:=0;
            end;
            writeln(kq);
            readln;
        End.
