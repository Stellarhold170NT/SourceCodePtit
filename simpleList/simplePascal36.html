              Uses Crt;
        Var     s:string;
                n,i:integer;
        Begin
                readln(s);
                readln(n);
                for i:=1 to length(s) do
                begin
                        if (ord(s[i])-n)>=97 then
                                write(chr(ord(s[i])-n))
                        else
                                write(chr(122-(n-ord(s[i])+97)+1));
                end;
        End.
