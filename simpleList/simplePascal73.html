              Uses Crt,math;
        Var n,w,i,j,min:longint; kq:int64; a:array[1..1000] of longint;
        Begin
            readln(n,w); min:=maxlongint; 
            for i:=1 to n do
            begin
                read(a[i]);
                if a[i]<min then min:=a[i];
                if (a[i]>=min) then kq:=max(kq,(w div min)*(a[i]-min));
            end;
        
            writeln(kq);
            readln;
        End.
