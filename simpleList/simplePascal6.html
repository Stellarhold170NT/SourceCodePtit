              USES CRT;
        var     s:ansistring;
                res,t:int64;
                i:longint;
                begin
                    readln(s);
                    res:=0;
                    for i:=1 to length(s) do
                        inc(res,ord(s[i])-48);
                    while (res>9) do begin
                        t:=0;
                        while (res>0) do begin
                            inc(t,res mod 10);
                            res:=res div 10;
                        end;
                        res:=t;
                    end;
                    writeln(res);
                end.
