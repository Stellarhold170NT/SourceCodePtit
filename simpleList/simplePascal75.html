              Uses Crt;
        Var n,k,i,r,j:longint; a:array[1..100000] of longint;
        Begin
            readln(n,k);
            for i:=1 to n do read(a[i]);
            i:=a[k] mod n;
            r:=a[k] div n; //writeln(r);
            a[k]:=0;
            j:=k;
            while i>0 do
            begin
                inc(j);
                if j>n then j:=1;
                a[j]:=a[j]+1;
                dec(i);
            end;
            //for i:=1 to n do write(a[i],' ');  writeln;
            for i:=1 to n do write(a[i]+r,' ');
            readln;
        End.
