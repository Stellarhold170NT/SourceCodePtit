              #define _CRT_SECURE_NO_WARNINGS
        #include <iostream>
        #include <cstring>
        using namespace std;
        
        int n, m;
        char arr[101][101];
        
        void dfs(int x, int y) {
          arr[x][y] = '.';
          if (x + 1 < n) {
            if (arr[x + 1][y] == '#') dfs(x + 1, y);
          }
          if (x - 1 >= 0) {
            if (arr[x - 1][y] == '#') dfs(x - 1, y);
          }
          if (y + 1 < m) {
            if (arr[x][y + 1] == '#') dfs(x, y + 1);
          }
          if (y - 1 >= 0) {
            if (arr[x][y - 1] == '#') dfs(x, y - 1);
          }
        }
        
        int main() {
        
          cin >> n >> m;
        
        
        
          for (int i = 0; i < n; i++) {
            string s;
            cin >> s;
            strcpy(arr[i], s.c_str());
          }
        
          int res = 0;
          for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
              if (arr[i][j] == '#') {
                res++;
                dfs(i, j);
              }
            }
          }
        
          cout << res << endl;
        
          return 0;
        }
