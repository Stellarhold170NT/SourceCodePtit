              Uses Crt,Math;
      Var n,k,i,j,kq:longint;
          a,t,f:array[-1001..1001,-1001..1001] of longint;
      Begin
          readln(n,k);
          for i:=1 to n do
           for j:=1 to n do read(a[i,j]);
          kq:=-maxlongint;
          for i:=1 to n do
           for j:=1 to n do
            begin
                t[i,j]:=t[i,j-1]+a[i,j]-a[i,j-k];
                f[i,j]:=f[i-1,j]+t[i,j]-t[i-k,j];
                kq:=max(kq,f[i,j]);
            end;
          writeln(kq);
          readln;
      End.
