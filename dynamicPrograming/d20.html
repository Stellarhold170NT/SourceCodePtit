              #include <iostream>
      using namespace std;
      
      int main() {
        int c, n;
        cin >> c >> n;
      
        int arr[100];
        for (int i = 0; i < n; i++) {
          cin >> arr[i];
        }
      
        int f[100000]{ 0 };
        f[0] = 1;
      
        for (int i = 0; i < n; i++) {
          for (int j = c; j >= arr[i]; j--) {
            if (j >= arr[i] && f[j - arr[i]] == 1) {
              f[j] = 1;
            }
          }
        }
      
        for (int j = c; j >= 0; j--) {
          if (f[j] == 1) {
            cout << j << endl;
            break;
          }
        }
      
        return 0;
      }
