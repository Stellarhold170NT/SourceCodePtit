              #include <iostream>
      using namespace std;
      
      int main() {
        int n;
        cin >> n;
      
        int* c = new int[n + 1];
      
        for (int i = 1; i <= n - 1; i++) {
          cin >> c[i];
        }
      
        long long* f = new long long[n + 1];
      
        f[1] = 0;
        f[2] = c[1];
        if (n - 1 > 1) f[3] = c[1] + c[2];
        for (int i = 4; i <= n; i++) {
          f[i] = min(f[i - 1] + c[i - 1], f[i - 2] + c[i - 1]);
        }
      
        cout << f[n] << endl;
      
        return 0;
      
      }
