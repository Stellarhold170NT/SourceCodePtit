              #include <bits/stdc++.h>
      using namespace std;
      int main()
      {
          int n, a[100000];
          cin>>n;
          cin>>a[1]>>a[2];
          int m=a[1], res=a[2]-a[1];
          if (a[2]<m) m=a[2];
          for (int i=3; i<=n; i++)
          {
              cin>>a[i];
              if (res<a[i]-m) res=a[i]-m;
              if (a[i]<m) m=a[i];
          }
          cout<<res;
      }
