              Uses Crt,Math;
      Var n,i:longint;
          a,u,v,h:array[-100..100001] of int64;
      Begin
          readln(n);
          for i:=1 to n do read(a[i]);
          u[0]:=-maxlongint;
          v[0]:=-maxlongint;
          h[0]:=-maxlongint;
          for i:=1 to n do
           begin
               u[i]:=max(u[i-1],a[i]);
               v[i]:=max(u[i-1]+a[i]*2,v[i-1]);
               h[i]:=max(v[i-1]+a[i]*3, h[i-1]);
           end;
          writeln(h[n]);
          readln;
      End.
