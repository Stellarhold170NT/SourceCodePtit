          readln(n,k);

          begin
              read(num);
              a[num]:=true;
          end;
          dem:=1;
          for i:=1 to n do
          begin
              d:=i; c:=i;
              if dem>n-i+1 then break;
              
              while a[c] do inc(c);
              if a[0] then
              begin
                  inc(c);
                  while a[c] do inc(c);
              end;

          end;
