              #include <iostream>
      using namespace std;
       
      int main() {
        int t;
        cin >> t;
       
        long long f[26][26];
       
        for (int i = 0; i <= 25; i++) {
          for (int j = 0; j <= 25; j++) {
            f[i][j] = 0;
          }
        }
       
        for (int j = 1; j <= 25; j++) {
          f[j][1] = 1;
        }
       
        for (int i = 1; i <= 25; i++) {
          for (int j = 2; j <= 25; j++) {
            f[i][j] = f[i - 1][j - 1] + j * f[i - 1][j];
          }
        }
       
       
        while (t--) {
          int n, k;
          cin >> n >> k;
          cout << f[n][k] << endl;
        }
       
        return 0;
      }
