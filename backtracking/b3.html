              Uses Crt;
        Var k,n,i,dem:longint;
            x:array[1..10] of longint;
        Procedure KN(i:longint);
        var j,l:longint;
        begin
            for j:=1 to k do
             begin
                 x[i]:=j;
                 if i=n then
                  begin
                      for l:=1 to n do write(x[l]);
                      writeln
                  end
                  else
                  KN(i+1);
             end;
        end;
        Begin
            readln(k,n);
            dem:=1;
            for i:=1 to n do dem:=dem*k;
            writeln(dem);
            KN(1);
            readln;
        End.
